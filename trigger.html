$(function() {

	var ft = $.filtrify("container", "placeHolder");

	$("span#thriller").click(function() {
		ft.trigger({ genre : ["Thriller"] });
	});

	$("span#comedy").click(function() {
		ft.trigger({ genre : ["Comedy"] });
	});

	$("span#thrillerComedy").click(function() {
		ft.trigger({ genre : ["Thriller", "Comedy"] });
	});

	/* Use JSON when you have categories with two or more words */
	$("span#thrillerComedyTomHanks").click(function() {
		ft.trigger({ 
			"genre" : ["Thriller", "Comedy"],
			"main actors" : ["Tom Hanks"]  
		});
	});

	$("div#triggers > span").click(function() {
		$(this)
			.addClass("selected")
			.siblings("span")
			.removeClass("selected");
	});

});
